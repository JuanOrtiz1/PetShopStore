CREATE TABLE `sac`.`sac_tb_product` (
  `PRODUCT` INT NOT NULL AUTO_INCREMENT,
  `DESCRIPTION` VARCHAR(100) NOT NULL,
  `SUPPLIER` VARCHAR(50) NOT NULL,
  `HIGH` INT NOT NULL,
  `TYPE` VARCHAR(100) NOT NULL,
  PRIMARY KEY (`PRODUCT`))
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8
COLLATE = utf8_spanish_ci
COMMENT = 'Tabla para listar los productos que existen en el punto de venta.';

CREATE TABLE `sac`.`sac_tb_supplier` (
  `SUPPLIER` INT NOT NULL AUTO_INCREMENT,
  `DESCRIPTION` VARCHAR(100) NOT NULL,
  `NUMBER_PHONE` VARCHAR(50) NOT NULL,
  `EMAIL` VARCHAR(100) NOT NULL,
  PRIMARY KEY (`SUPPLIER`))
)
ENGINE = InnoDB
DEFAULT CHARACTER SET = utf8
COLLATE = utf8_spanish_ci
COMMENT = 'Tabla de proveedores.';

ALTER TABLE sac.sac_tb_supplier ADD COLUMN STATUS_SUPPLIER CHAR(1);



CREATE TABLE `sac`.`sac_tb_product` (
  `PRODUCT` INT AUTO_INCREMENT PRIMARY KEY,
  `NAME_PRODUCT` VARCHAR(255),
  `CATEGORY` VARCHAR(255),
  `PRESENTATION` VARCHAR(50),
  `PARAM_SUPPLIER` INT,
  FOREIGN KEY (`PARAM_SUPPLIER`) REFERENCES `sac_tb_supplier`(`SUPPLIER`),
  `PRODUCT_WEIGHT` INT(7),
  `PRODUCT_WEIGHT_TYPE` INT(7),
  `PURCHASE_PRICE` INT(7),
  `SALE_PRICE` INT(7),
  `REVENUE` INT(7)
);
COMMENT = 'Tabla de productos';
